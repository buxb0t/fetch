Dim fso, sourceFile, startupFolder, wshShell Set fso = CreateObject("Scripting.FileSystemObject") Set wshShell = CreateObject("WScript.Shell") sourceFile = WScript.ScriptFullName startupFolder = wshShell.SpecialFolders("Startup") & "\sayHi.vbs" If Not fso.FileExists(startupFolder) Then fso.CopyFile sourceFile, startupFolder End If wshShell.Popup "Hi", 1, "Message", 64 wshShell.Run "wscript.exe """ & startupFolder & """", 0, False
